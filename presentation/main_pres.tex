\section{Setup}


\begin{frame}{Setting geometrico}
\begin{itemize}
  \item Lo spazio di Minkowski è $\M=\R^3$ con pseudometrica
  \[
    ds^2 = (dx^1)^2+(dx^2)^2-(dx^3)^2.
  \]
  \item Una superficie immersa $ \iota:S\hookrightarrow \M$ è \alert{space-like} se la metrica indotta è Riemanniana.
  \item Prima forma fondamentale → si calcola come  
  \item Una superficie space-like completa è, in particolare, un grafico intero su $\R^2$. Dunque è semplicemente connessa.
  \item Fissato un valore per la curvatura media $\kappa$ è possibile mettere in biezione tramite $L:\M\to \M: x\mapsto \kappa x$,
\begin{align*}
	\left\{
	\begin{array}{c}
		\text{superfici space-like complete}\\
		\text{a curvatura costante }\kappa
	\end{array}
	\right\}
	\quad \overset{L}{\longleftrightarrow}\quad
	\left\{
	\begin{array}{c}
		\text{superfici space-like complete}\\
		\text{a curvatura costante }1
	\end{array}
	\right\}
\end{align*}


\end{itemize}

\vspace{1mm}
\begin{block}{Definizione}
	Diciamo taglio CMC una superficie space-like completa e con curvatura media costante.
\end{block}

\end{frame}

\begin{frame}{Parametrizzazione conforme}

\begin{block}{Struttura complessa e uniformizzazione}
	\begin{itemize}
		\item Fissate $(x,y)$ coordinate isoterme che preservano l'orientazione, $z=x+iy$ determina una struttura complessa.
		\item Se inoltre $S$ è semplicemente connessa, il teorema di uniformizzazione implica l'esistenza di un diffeomorfismo conforme
\[
\sigma:\Sigma \to S,
\qquad
\Sigma \in \{\C,\widehat{\C},\mathbb{D}\}.
\]
	\end{itemize}
\end{block}

\vspace{1mm}
$S$ è iperbolica per ipotesi, allora
\[
\Sigma=\mathbb{D},
\]
dotata della metrica di Poincaré
\[
ds_p^2=\frac{4}{(1-|z|^2)^2}|dz|^2.
\]
Possiamo quindi leggere $S$ come
\[
\sigma:\mathbb{D}\to S.
\]
\end{frame}

\begin{frame}{Risultato principale}
Consideriamo $\EE$ lo spazio delle superfici CMC-$1$ iperboliche a meno di isometria, e $\QDD$ lo spazio dei differenziali quadratici olomorfi su $\D$, a meno di cambio di coordinate conformi.

\vspace{2mm}
Prendiamo l'applicazione
\[
\Amap:\EE\longrightarrow \QDD
\]
che associa a una superficie il differenziale quadratico ottenuto dalla mappa di Gauss.

\begin{block}{Teorema}
L'applicazione $\Amap$ è ben definita e biiettiva.
\end{block}

\vspace{1mm}
Procediamo a:
\begin{enumerate}
  \item illustrare la costruzione di $\Amap$;
  \item illustrare come la biettività si riduce ad esistenza e unicità per una certa PDE;
  \item illustrare il risultato di esistenza e unicità.
\end{enumerate}
\end{frame}

\section{Il risultato di classificazione, la mappa $\Amap$}

\begin{frame}{La mappa di Gauss}
\begin{itemize}
	\item Se $S\subset \M$ è una superficie space-like orientata, il suo vettore normale unitario è time-like.
	
	\item Quindi la normale unitaria prende valori nell'iperboloide
	\[
	\HH=\{x\in \M:\langle x,x\rangle=-1,\ x^3>0\} \cong \mathbb{D}.
	\]
	
	\item Questo permette di definire la mappa di Gauss
	\[
	N:S\to \HH:p\mapsto N_p \quad \text{oppure}\quad u = j\circ N\circ \sigma :\mathbb{D}\to \mathbb{D}
	\]
	
	\item Risultato fondamentale:
	
\end{itemize}

	
	\begin{block}{Teorema}
		Una superficie space-like ha curvatura media costante se e solo se la sua mappa di Gauss è armonica.
	\end{block}
\vspace{2mm}

\end{frame}



\begin{frame}[t]{Il differenziale di Hopf}

\begin{columns}[T,onlytextwidth]
	\column{0.48\textwidth}
	\begin{block}{ Decomposizione tensori tipo 0-2}
		Dato $\omega$ tensore $(0,2)$ su $S$, è possibile decomporre $\omega$ in
		\begin{align*}
			\omega = \omega^{(2,0)} + \omega^{(1,1)} + \omega^{(0,2)}.
		\end{align*}

		\vspace{-2mm}

		con
		\begin{itemize}\itemsep0.4em
			\item $\omega^{(2,0)}$ e $\overline{\omega^{(0,2)}}$ localmente della forma
				$\varphi(z)\,dz^2$,
			\item $\omega^{(1,1)}$ localmente della forma $k(z)|dz|^2$
		\end{itemize}
			Diciamo che $\varphi(z) dz^2$ è olomorfo se $\varphi$ è olomorfa come funzione a valori in $\C$.
	\end{block}
	\column{0.48\textwidth}
	\vspace{-2mm}
	\textbf{Applicato a $u^*(g_{\mathbb D})$} si ottiene
	\begin{align*}
		u^*(g_\D)=\Phi+\mathcal H|dz|^2+\overline{\Phi}.\quad\Phi := \Hopf(u).
	\end{align*}
	$u$ è la mappa di Gauss di $S$ ed è armonica. Allora:
	\begin{itemize}\itemsep0.4em
		\item $\Phi$ è un differenziale quadratico olomorfo,
		\item $\mathcal H|dz|^2$ è la metrica indotta su $S$.
	\end{itemize}

	Possiamo scrivere
	\begin{align*}
		\tilde w=\frac12\log\mathcal H(u),
	\end{align*}
	e quindi
	\begin{align*}
		\I=e^{2\tilde w}|dz|^2=e^{2w}ds_p^2.
	\end{align*}
\end{columns}

\end{frame}



\begin{frame}{La mappa \boldmath{$\mathcal{A}$}}
	\begin{block}{}
		Riassumendo, abbiamo costruito l'associazione
		\[
			S \longmapsto u \longmapsto (\Phi,w) \longmapsto \Phi.
		\]
		Con $S$ superficie CMC-1, $u$ mappa di Gauss letta in carta, $\Phi = \Hopf(u)$ e $w = \frac{1}{2}\mathcal{H}(u)$.
	\end{block}
	\vspace{1mm}
	Dove
	\begin{itemize}
		\item $u$ è armonica,
		\item $\Phi$ è olomorfo,
		\item $\Iff=e^{2w}ds_p^2$ è la metrica della superficie letta su $\D$.
	\end{itemize}

	\vspace{1mm}
	L'applicazione $\mathcal{A}$ si costruisce quozientando
	\[
		\Amap:\EE\to \QDD : [S] \mapsto [\Phi].
	\]
\end{frame}


\begin{frame}{Invertire \boldmath{$\mathcal{A}$}}
	
Partiamo da un differenziale quadratico olomorfo $\Phi\in\QDD,$ e cerchiamo di ricostruire la superficie $S$.
\begin{block}{Teorema (Bonnet)}
	Date una forma bilineare simmetrica definita positiva $g$, e una forma bilineare simmetrica $h$
	su un dominio semplicemente connesso di $\R^2$,  se $g$ e $h$ soddisfano le equazioni di Gauss-Codazzi per $\M$, allora esiste una superficie space-like immersa in $\M$ con forme fondamentali $g$ e $h$, unica a meno di isometria di $\M$.
\end{block}

\vspace{1mm}

Per applicarlo, dobbiamo quindi
\begin{enumerate}
	\item la metrica indotta $\Iff$;
	\item la seconda forma fondamentale $\IIff$;
	\item verificare che $\Iff$ e $\IIff$ soddisfino le equazioni di Gauss-Codazzi.
\end{enumerate}

\vspace{1mm}

\end{frame}

\begin{frame}[t]{La prima forma fondamentale e l'equazione di Gauss}
  Dato $\Phi\in\QDD$, cerchiamo la metrica di $S$
  nella forma
  \[
    \Iff \;=\; e^{2w}\,ds_p^2,\qquad w:\D\to\R\ \text{incognita}.
  \]

  \vspace{1mm}
  \begin{columns}[T,onlytextwidth]
    \column{0.14\textwidth}\hfill
    \column{0.72\textwidth}
    \begin{block}{Riformulazione dell'equazione di Gauss}
      \centering
      L'equazione di Gauss per $\Iff$ è equivalente alla PDE
      \[
        \Delta_p w \;=\; e^{2w}-\norm{\Phi}^2 e^{-2w}-1 \quad\text{su }\D.
      \]
    \end{block}
    \column{0.14\textwidth}\hfill
  \end{columns}

  \vspace{2mm}
  Inoltre valgono anche i seguenti fatti
  \begin{itemize}\itemsep=2pt
    \item Rappresentando la metrica di $S$ come $\I=e^{2w}\,ds_p^2$, si richiede che essa sia completa 
    \item La mappa di Gauss di $S$ preserva l'orientazione si traduce in $J(u)>0$ e dunque in
          $\;e^{2w}-\norm{\Phi}^2 e^{-2w}>0$.
  \end{itemize}
\end{frame}

\begin{frame}[t]{Ricostruzione della seconda forma fondamentale}
Se $\II$ è la seconda forma fondamentale di $S$ e $\II_0$ la sua parte traceless,
si ha
\[
2\II_0 = \Hopf(N) + \overline{\Hopf(N)},
\qquad
\tr_\I(\II)=2H.
\]

\vspace{1mm}

Se $\Hopf(N) = \Phi = \varphi(z) dz^2$, e $\II = h_{ij}dx^idx^j$si ottengono quindi le relazioni:
\vspace{1mm}

\begin{center}
\begin{minipage}{0.72\textwidth}
\begin{block}{Condizioni necessarie su $(h_{ij})_{ij}$}
\[
\begin{cases}
h_{11}+h_{22}=2e^{2\tilde w}\\
h_{11}-h_{22}=2\Re(\varphi)\\
h_{12}=-\Im(\varphi)\ .
\end{cases}
\]
\end{block}
\end{minipage}
\end{center}

\vspace{1mm}

Dati \(\tilde w\) e \(\Phi\) possiamo ricostruire $\II$ usando il sistema.
\end{frame}


\begin{frame}[t]{L'equazione di Codazzi}
Sia $g=e^{2\widetilde w}|dz|^2$ una metrica conforme su $\D$ e sia $h$ una
2-forma bilineare simmetrica reale, di parte traceless $h_0$.

\vspace{1mm}

\begin{itemize}
  \item Sapendo che $H$ è costante si ha
  \[
    h \text{ verifica Codazzi}
    \iff
    h_0 \text{ verifica Codazzi}.
  \]

  \item Essendo $h$  a valori reali, si può scrivere
  \[
    h_0=\Phi+\overline\Phi.
  \]

  \item Si ha:
  \[
    h_0=\Phi+\overline\Phi \text{ verifica Codazzi rispetto a} g
    \iff
    \Phi \text{ è olomorfo}.
  \]
\end{itemize}

\vspace{1mm}

Si deduce quindi che, per la candidata seconda forma fondamentale
\[
\IIff := e^{2\widetilde w}|dz|^2 + (\Phi+\overline\Phi),
\]
vale l'equazione di Codazzi, essendo $\Phi$ olomorfo per ipotesi.
\end{frame}



\begin{frame}[t]{Riassumendo}
	Finora abbiamo dimostrato che $\Amap$ è \alert{suriettiva} seguendo lo schema

  \vspace{2mm}
  \begin{center}

      \draw[|->] (phi)     -- node[above,lbl] {} (normphi);%(1)

  In \astr{} richiediamo sia possibile trovare $w$ tale che $\I$ soddisfa l'equazione di Gauss.\\
  \vspace{6mm}
  Per quanto riguarda l'\alert{iniettività} di $\Amap$, si mostra che se due superfici hanno lo stesso $\Phi$ allora sono isometriche, richiedendo l'unicità del passaggio in \astr{} e il teorema di Bonnet.
\end{frame}


\begin{frame}[t]{Il problema analitico}
	Abbiamo ridotto il teorema di classificazione alla risoluzione del seguente problema

  \begin{columns}[T,onlytextwidth]
    \column{0.08\textwidth}\hfill
    \column{0.84\textwidth}
    \begin{block}{ Problema }
      Dato $\Phi\in\QDD$, esiste \alert{un'unica}
      $w\in C^\infty(\D)$ che verifica
      \[
        \Delta_p w \;=\; e^{2w}-\norm{\Phi}^2 e^{-2w}-1
        \quad\text{su } \D,
      \]
	  e tale che
      \begin{itemize}\itemsep=2pt
        \item $e^{2w}\,ds_p^2$ è una metrica completa su $\D$;
        \item $e^{2w}-\norm{\Phi}^2 e^{-2w} > 0$ su $\D$.
      \end{itemize}
    \end{block}
    \column{0.08\textwidth}\hfill
  \end{columns}

  \vspace{3mm}
  \begin{itemize}\itemsep=2pt
    \item \alert{unicità}: principio del massimo generalizzato;
    \item \alert{esistenza}: metodo delle sotto-soluzioni/sovra-soluzioni.
  \end{itemize}
\end{frame}




\section{Il problema analitico}


\begin{frame}[t]{Unicità: principio del massimo generalizzato}
  La PDE è ambientata su $(\D,ds_p^2)$.

  \vspace{1mm}
  \begin{columns}[T,onlytextwidth]
    \column{0.07\textwidth}\hfill
    \column{0.86\textwidth}
    \begin{block}{Principio del massimo generalizzato}
      Sia $(M,g)$ Riemanniana completa con la curvatura di Ricci limitata dal basso, e sia $f\in C^2(M)$ limitata superiormente. Allora esiste una successione di punti $(p_n)\subset M$ tale che
      \[
        f(p_n)\to \sup_M f,\qquad
        |\nabla_g f|(p_n)\to 0,\qquad
        \limsup_n \Delta_g f(p_n)\le 0.
      \]
    \end{block}
    \column{0.07\textwidth}\hfill
  \end{columns}

  \vspace{1mm}
  Nella dimostrazione di unicità:
  \begin{itemize}
    \item si fissano due soluzioni \(v,\widetilde w\);
    \item si applica il principio alla differenza \(\eta=v-\widetilde w\), verificando che $\sup \eta = \overline{\eta} \leq 0$;
    \item scambiando i ruoli di \(v\) e \(\widetilde w\), si conclude \(v=\widetilde w\).
  \end{itemize}
\end{frame}
\begin{frame}[t]{Esistenza: sotto/sovra-soluzioni ed esaustione}

\begin{block}{Metodo delle sotto/sovra-soluzioni}
Sia \(M\) una varietà Riemanniana completa e sia \(F(x,u)\in C^\infty(M\times\R)\) con \(\partial_uF>0\).
Se esistono \(\underline\psi,\overline\psi\in C^0(M)\cap H^1(M)\) con \(\underline\psi\le \overline\psi\) tali che, debolmente,
\(\Delta \overline\psi \le F(x,\overline\psi)\) e \(\Delta \underline\psi \ge F(x,\underline\psi)\),
allora esiste \(u\in C^\infty(M)\) tale che \(\Delta u=F(x,u)\) e \(\underline\psi\le u\le \overline\psi\).
\end{block}

\vspace{1mm}

Nel nostro caso:
\begin{itemize}

  \item nel caso generale, si costruisce una successioni di limitati $(D_k)_k$ che invade \(\D\). Su ciascun dominio $\|\Phi\|$ è limitata.

  \item Dal teorema di unicità si ottiene che \((w_k)_k\) è monotona. Da questo si trova poi la convergeza in $C^2_{\mathrm{loc}}$.
\end{itemize}

\end{frame}
